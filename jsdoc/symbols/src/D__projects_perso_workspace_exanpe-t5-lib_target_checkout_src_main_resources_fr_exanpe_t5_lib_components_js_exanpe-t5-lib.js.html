 43 /** 
 44  * Constructor
 45  * @class Represents an Accordion. Can be accessed through JavaScript by its id.
 46  * @param {String} id the id of the main accordion element
 47  * @param {boolean} multiple if multiple items can be opened at once
 48  * @param {String} eventType the type of event that trigger the accordion
 49  * @param {float} duration the speed of accordion animation
 50  */
 51 Exanpe.Accordion = function(id, multiple, eventType, duration){
 52 	
 53 	/**
 54 	 * id of the accordion
 55 	 */
 56 	this.id = id;
 57 	
 58 	/**
 59 	 * Multiple item visible feature
 60 	 */
 61 	this.multiple = multiple;
 62 	
 63 	/**
 64 	 * Shows on item on over instead of click
 65 	 */
 66 	this.eventType = eventType;
 67 	
 68 	/**
 69 	 * Duration for animation
 70 	 */
 71 	this.duration = duration;
 72 	
 73 	/**
 74 	 * Current item opened
 75 	 */
 76 	this.currentItem = null;
 77 	
 78 	/**
 79 	 * Marker defining if an animation is on
 80 	 */
 81 	this.animating = false;
 82 };

109 /**
110  * Action triggered on an item
111  * @param {Event} e the event
112  * @private
113  */
114 Exanpe.Accordion.prototype._action = function(e){
115 	
116 	var itemId = YAHOO.util.Dom.getAncestorByTagName(YAHOO.util.Event.getTarget(e), "li").id;
117 	
118 	if(this.isOpened(itemId)){
119 		if(this.eventType == "click"){
120 			this.close(itemId);
121 		}
122 	}
123 	else{
124 		this.open(itemId);
125 	}
126 	return false;
127 };
128 
129 /**
130  * Open an item in this accordion
131  * @example accordionId.open("item1")
132  * @param {string} id the id of the item of the accordion to open
133  * @returns {boolean} true if open occurred, false otherwise
134  */
135 Exanpe.Accordion.prototype.open = function(id){
136 	if(this.animating){
137 		return false;
138 	}
139 	
140 	if(this.isOpened(id)){
141 		return false;
142 	}
143 	
144 	if(!this.multiple && this.currentItem && this.currentItem != id){
145 		this.close(this.currentItem);
146 	}
147 	
148 	if(!this.multiple){
149 		this.currentItem = id;
150 	}
151 	
152 	var contentEl = this.getContentEl(id);
153 	
154 	//open to get height : fail if hidden
155 	YAHOO.util.Dom.setStyle(contentEl,"display","block");
156 	
157 	var height = this.getContentElHeight(contentEl);
158 	
159 	YAHOO.util.Dom.setStyle(contentEl,"visibility","visible");
160 	YAHOO.util.Dom.setStyle(contentEl,"height","1px");
161 	
162 	var atts = {
163 			height : {from : 1, to : height}
164 	};
165 
166 	var acc = this;
167 	var anim = new YAHOO.util.Anim(contentEl, atts, this.duration);
168 	anim.onComplete.subscribe(function(){
169 		YAHOO.util.Dom.setStyle(contentEl,"display","");
170 		acc.animating = false;
171 	});
172 	
173 	
174 	YAHOO.util.Dom.removeClass(id, Exanpe.Accordion.CSS_CLOSED);
175 	YAHOO.util.Dom.addClass(id, Exanpe.Accordion.CSS_OPENED);
176 	
177 	this.animating = true;
178 	anim.animate();
179 	
180 	return true;
181 };

184 /**
185  * Hides an item
186  * @param {string} id the id of the item to hide
187  * @returns {boolean} true if close occurred, false otherwise
188  */
189 Exanpe.Accordion.prototype.close = function(id){
190 	if(this.animating){
191 		return false;
192 	}
193 	
194 	if(!this.isOpened(id)){
195 		return false;
196 	}
197 	
198 	if(!this.multiple && this.currentItem != id){
199 		return false;
200 	}
201 		
202 	var contentEl = this.getContentEl(id);
203 	
204 	var height = this.getContentElHeight(contentEl);
205 	
206 	var acc = this;
207 	var anim = new YAHOO.util.Anim(contentEl, Exanpe.Accordion.HIDE_ANIM_ATT, this.duration);
208 	anim.onComplete.subscribe(function(){
209 		YAHOO.util.Dom.removeClass(id, Exanpe.Accordion.CSS_OPENED);
210 		YAHOO.util.Dom.addClass(id, Exanpe.Accordion.CSS_CLOSED);
211 		YAHOO.util.Dom.setStyle(contentEl,"height",height+"px");
212 		acc.animating = false;
213 	});
214 
215 	this.animating = true;
216 	anim.animate();
217 	
218 	this.currentItem = null;
219 	
220 	return true;
221 };

252 /**
253  * Get the content element from the item id

302 /** 
303  * Constructor
304  * @class Represents an Hideable Panel. Can be accessed through JavaScript by its id.
305  * @param {String} id the id of the main accordion element
306  * @param {int} hideWidth the width of the hideable panel
307  * @param {float} duration the speed of hidding animation
308  */
309 Exanpe.HideablePanel = function(id, hideWidth, duration){
310 	/**
311 	 * id of the instance
312 	 */
313 	this.id = id;
314 	
315 	/**
316 	 * duration of the animation
317 	 */
318 	this.duration = duration;
319 	
320 	/**
321 	 * state of the panel
322 	 */
323 	this.hidden = false;
324 	
325 	var attributesHide = { 
326         width: { to: 0 }
327     }; 
328 	this.animHide = new YAHOO.util.Anim(this.id+'_hidepart', attributesHide, this.duration);
329 	this.animHide.onComplete.subscribe( function(){
330 		YAHOO.util.Dom.addClass(id, 'hidden');
331 	});
332 	
333 	var attributesShow = { 
334         width: { to: hideWidth  ,unit: 'px'}
335     }; 
336 	this.animShow = new YAHOO.util.Anim(this.id+'_hidepart', attributesShow, this.duration); 
337 	this.animShow.onComplete.subscribe( function(){
338 		YAHOO.util.Dom.removeClass(id, 'hidden');
339 	});
340 	
341 	var hideBar = this.getHideBarEl();
342 	
343 	YAHOO.util.Event.addListener(hideBar, "click", Exanpe.HideablePanel.prototype.reverse, null, this);
344     YAHOO.util.Event.addListener(hideBar, "mouseover", function(ev, id){YAHOO.util.Dom.addClass(hideBar, 'hidebarHover');}, this.id);
345     YAHOO.util.Event.addListener(hideBar, "mouseout", function(ev, id){YAHOO.util.Dom.removeClass(hideBar, 'hidebarHover');}, this.id);
346 };
347 
348 /**
349  * Method to get the DOM element of the hide bar
350  * @return {HTMLElement} the hidding bar
351  * @private 
352  */
353 Exanpe.HideablePanel.prototype.getHideBarEl = function(){
354 	return YAHOO.util.Dom.get(this.id+"_hidebar");
355 };
356 
357 /**
358  * Reverse the visibility of the panel
359  * @return {boolean} true if the panel gets visible, false if it gets hidden
360  */
361 Exanpe.HideablePanel.prototype.reverse = function(){
362 	if(this.hidden){
363 		this.hidden = false;
364 		this.animShow.animate();
365 		return true;
366 	}else{
367 		this.hidden = true;
368 		this.animHide.animate();
369 		return false;
370 	}	
371 };
372 
373 /**
374  * Returns the visibility of the panel
375  * @return {boolean} true if the panel is visible, false otherwise
376  */
377 Exanpe.HideablePanel.prototype.isVisible = function(){
378 	return !this.hidden;
379 };

444  * @static

449 /**
450  * Method canceling a hover in progress
451  * @param {Event} e the event of cancel
452  * @private
453  */
454 Exanpe.SecurePassword.prototype._cancelHover = function(e){
455 	window.clearTimeout(this.timeoutHover);
456 	this.timeoutHover = null;
457 };
458 
459 /**
460  * Method triggered on hover start
461  * @param {Event} e the event of hovering
462  * @private
463  */
464 Exanpe.SecurePassword.prototype._hover = function(e){
465 
466 	var value = this.positionnings[YAHOO.util.Event.getTarget(e).id];
467 	
468 	YAHOO.util.Event.addListener(YAHOO.util.Event.getTarget(e).id, "mouseout", Exanpe.SecurePassword.prototype._cancelHover, this, true);
469 	if(this.timeoutHover){
470 		this._cancelHover();
471 	}
472 	
473 	var obj = this;
474 	
475 	this.timeoutHover = window.setTimeout(function(){
476 		obj._processRecording(value);
477 		obj = null;
478 		value = null;
479 	}, Exanpe.SecurePassword.HOVER_WAIT);	
480 	
481 };
482 
483 /**
484  * Method triggered on click
485  * @param {Event} e the event of hovering
486  * @private
487  */
488 Exanpe.SecurePassword.prototype._click = function(e){
489 	var value = this.positionnings[YAHOO.util.Event.getTarget(e).id];
490 	this._processRecording(value);
491 };

494 /**
495  * Method actually performing the record of a value
496  * @param {char} value the character to record
497  * @private
498  */
499 Exanpe.SecurePassword.prototype._processRecording = function(value){
500 	var inputEl = this.getInputEl();
501 	var pwdEl = this.getHiddenEl();
502 	
503 	if(!inputEl.value){
504 		inputEl.value = value;
505 		pwdEl.value = inputEl.value;
506 		return;
507 	}
508 	
509 	if(inputEl.size && inputEl.value.length >= inputEl.size){
510 		return;
511 	}
512 	
513 	inputEl.value = inputEl.value + value;
514 	pwdEl.value = inputEl.value;
515 	pwdEl = null;
516 	inputEl = null;
517 };

573 /** 
574  * Constructor
575  * @class Represents a Slider. Can be accessed through JavaScript by its id.
576  * @param {String} id the id of the slider element
577  * @param {int} min the minimum slider value
578  * @param {int} max the maximum slider value
579  * @param {String} displayId the HTML element id to display the value, or null if no display
580  * @param {Boolean} ajax whether the Slider update is executed in Ajax mode, or not
581  * @param {String} url the url used to execute AJAX request
582  * @param {String} zone the Tapestry zone element to update at the end of the ajax request
583  */
584 Exanpe.Slider = function(id, min, max, displayId, ajax, url, zone){
585 	/**
586 	 * The id of the instance
587 	 */
588 	this.id = id;
589 	
590 	/**
591 	 * The minimum slider value
592 	 */
593 	this.min = min;
594 	
595 	/**
596 	 * The maximum slider value
597 	 */
598 	this.max = max;
599 	
600 	/**
601 	 * The HTML id element to display the value, or null
602 	 */
603 	this.displayId = displayId;
604 	
605 	/**
606 	 * Pixels number for 1 unit
607 	 */
608 	this.unit = Exanpe.Slider.LENGTH/(this.max-this.min);
609 	
610 	/**
611 	 * This wrapped yui slider instance
612 	 */
613 	this.yui = null;
614 	
615 	/**
616 	 * Is ajax mode or not
617 	 */
618 	this.ajax = ajax;
619 	
620 	/**
621 	 * Used to execute an AJAX request for updating the Slider
622 	 */
623 	this.url = url;
624 	
625 	/**
626 	 * Used to update a Tapestry zone element
627 	 */
628 	this.zone = zone;
629 	
630 	/**
631 	 * Current request
632 	 */
633 	this.request = null;
634 };

693 /**
694  * Set the value of the slider.
695  * @param {int} value the value to set
696  * @param {Boolean} skipSlider whether the slider value update should me skipped. Set to true if the value initially comes from the YUI component.
697  */
698 Exanpe.Slider.prototype.setValue = function(value, skipSlider) {
699 	if(value < this.min || value > this.max){
700 		Exanpe.Log.error("Incorrect value set:"+value+". Min:"+this.min+"/Max:"+this.max);
701 		return ;
702 	}
703 	if(this._getDisplayEl()){
704 		this._getDisplayEl().innerHTML = ""+value;
705 	}
706 	if(skipSlider !== true){
707 		this.yui.setValue(this._valueToPixel(value), true, false, true);
708 	}
709 	this.value = value;
710 	this._getHiddenEl().value = value;
711 };

715  * @return {int} the value of the slider
716  */
717 Exanpe.Slider.prototype.getValue = function(value) {
718 	return this.value;
719 };

721 /**
722  * Convert a value to a yui pixel value.
723  * @param {int} value the value to set
724  * @return a value to the pixel unit of YUI
725  * @private
726  */
727 Exanpe.Slider.prototype._valueToPixel = function(value) {
728 	var zero = (-this.min)*this.unit;
729 	return zero + value*this.unit;
730 };
731 
732 /**
733  * Return a value from the pixel of YUI
734  * @param {int} value the value to set
735  * @return a value from the pixel of YUI
736  * @private
737  */
738 Exanpe.Slider.prototype._pixelToValue = function(value) {
739 	var zero = (-this.min)*this.unit;
740 	return (value - zero)/this.unit;
741 };

761 /**
762  * Execute an Ajax request to update the slider value
763  * @private
764  */
765 Exanpe.Slider.prototype._ajaxSliderUpdate = function (){ 	
766 	
767 	var slider = this;
768 	
769 	// Ajax Failure handler
770 	var failureHandler = function(o){
771 		slider.request = null;
772 		Exanpe.Log.error("Could not process ajax Slider update");
773 	};
774 
775 	// Ajax Success handler
776 	var successHandler = function(o){
777 		if (slider.zone){

1454 	var highlight = this.highlight;
1455 	
1456 	// Target DIV to update

1458 		
1459 	// Ajax Failure handler
1460 	var failureHandler = function(o) 
1461 	{

1463 		Exanpe.Log.error("Could not process request in order to get the AjaxLoder content.");
1464 	};

1476 		if(highlight){
1477 			ajaxLoader._highlight(0, 1, 1);
1478 		}
1479 	};
1480 	
1481 	// Callback objects
1482 	var callback = 
1483 	{

1808 /**
1809  * Triggers the select loading 
1810  * @param {String} value the selected value
1811  */
1812 Exanpe.SelectLoader.prototype.executeRequest = function (value){
1813 	
1814 	var selectLoader = this;
1815 	
1816 	if(this.request){
1817 		YAHOO.util.Connect.abort(this.request);
1818 	}
1819 	
1820 	// Ajax Failure handler
1821 	var failureHandler = function(o) 
1822 	{
1823 		selectLoader.request = null;
1824 		Exanpe.Log.error("Could not get the SelectLoader target's content. Fail id:"+selectLoader.id);
1825 	};
1826 	
1827 	// Ajax Success handler
1828 	var successHandler = function(resp) 
1829 	{
1830 		selectLoader.request = null;
1831 		var content = resp.responseText;
1832 		
1833 		var targetSelect = document.getElementById(selectLoader.targetId);
1834 		
1835 		var xmlDoc;
1836 		if (window.DOMParser){
1837 		  parser=new DOMParser();
1838 		  xmlDoc=parser.parseFromString(content,"text/xml");
1839 		}else{ // Internet Explorer
1840 		  xmlDoc=new ActiveXObject("Microsoft.XMLDOM");
1841 		  xmlDoc.async="false";
1842 		  xmlDoc.loadXML(content);
1843         } 
1844 		
1845 		targetSelect.options.length = 0;
1846 		xmlDoc = xmlDoc.firstChild;//go to root
1847 		
1848 		for(var i=0;i<xmlDoc.childNodes.length;i++){
1849 
1850 			var cn = xmlDoc.childNodes[i];
1851 			if(cn.nodeType==1 && cn.nodeName=="option"){
1852 				var txt;
1853 				if(window.DOMParser){
1854 					txt = cn.textContent;
1855 				}else{
1856 					txt = cn.text;
1857 				}
1858 
1859 				var o = new Option(txt, cn.getAttribute("value"));
1860 				targetSelect.options.add(o);
1861 			}
1862 		}
1863 	};//end success
1864 	
1865 	// Callback objects
1866 	var callback = 
1867 	{
1868 		success: successHandler,
1869 		failure: failureHandler
1870 	};
1871 	
1872 	this.request = YAHOO.util.Connect.asyncRequest(
1873 		"GET",
1874 		this.url+"?"+Exanpe.SelectLoader.PARAM_NAME+"="+value, 
1875 		callback, 
1876 		null
1877 	);
1878 };
