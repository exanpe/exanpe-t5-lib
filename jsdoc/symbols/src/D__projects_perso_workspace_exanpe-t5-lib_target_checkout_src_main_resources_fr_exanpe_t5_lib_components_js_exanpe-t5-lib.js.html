 44  * Constructor
 45  * @class Represents an Accordion. Can be accessed through JavaScript by its id.
 46  * @param {String} id the id of the main accordion element
 47  * @param {boolean} multiple if multiple items can be opened at once
 48  * @param {String} eventType the type of event that trigger the accordion
 49  * @param {float} duration the speed of accordion animation
 50  */
 51 Exanpe.Accordion = function(id, multiple, eventType, duration){
 52 	
 53 	/**
 54 	 * id of the accordion
 55 	 */
 56 	this.id = id;
 57 	
 58 	/**
 59 	 * Multiple item visible feature
 60 	 */
 61 	this.multiple = multiple;
 62 	
 63 	/**
 64 	 * Shows on item on over instead of click
 65 	 */
 66 	this.eventType = eventType;
 67 	
 68 	/**
 69 	 * Duration for animation
 70 	 */
 71 	this.duration = duration;
 72 	
 73 	/**
 74 	 * Current item opened
 75 	 */
 76 	this.currentItem = null;
 77 	
 78 	/**
 79 	 * Marker defining if an animation is on
 80 	 */
 81 	this.animating = false;
 82 };

109 /**
110  * Action triggered on an item
111  * @param {Event} e the event
112  * @private
113  */
114 Exanpe.Accordion.prototype._action = function(e){
115 	
116 	var itemId = YAHOO.util.Dom.getAncestorByTagName(YAHOO.util.Event.getTarget(e), "li").id;
117 	
118 	if(this.isOpened(itemId)){
119 		if(this.eventType == "click"){
120 			this.close(itemId);
121 		}
122 	}
123 	else{
124 		this.open(itemId);
125 	}
126 	return false;
127 };

129 /**
130  * Open an item in this accordion
131  * @example accordionId.open("item1")
132  * @param {string} id the id of the item of the accordion to open
133  * @returns {boolean} true if open occurred, false otherwise
134  */
135 Exanpe.Accordion.prototype.open = function(id){
136 	if(this.animating){
137 		return false;
138 	}
139 	
140 	if(this.isOpened(id)){
141 		return false;
142 	}
143 	
144 	if(!this.multiple && this.currentItem && this.currentItem != id){
145 		this.close(this.currentItem);
146 	}
147 	
148 	if(!this.multiple){
149 		this.currentItem = id;
150 	}
151 	
152 	var contentEl = this.getContentEl(id);
153 	
154 	//open to get height : fail if hidden
155 	YAHOO.util.Dom.setStyle(contentEl,"display","block");
156 	
157 	var height = this.getContentElHeight(contentEl);
158 	
159 	YAHOO.util.Dom.setStyle(contentEl,"visibility","visible");
160 	YAHOO.util.Dom.setStyle(contentEl,"height","1px");
161 	
162 	var atts = {
163 			height : {from : 1, to : height}
164 	};
165 
166 	var acc = this;
167 	var anim = new YAHOO.util.Anim(contentEl, atts, this.duration);
168 	anim.onComplete.subscribe(function(){
169 		YAHOO.util.Dom.setStyle(contentEl,"display","");
170 		acc.animating = false;
171 	});
172 	
173 	
174 	YAHOO.util.Dom.removeClass(id, Exanpe.Accordion.CSS_CLOSED);
175 	YAHOO.util.Dom.addClass(id, Exanpe.Accordion.CSS_OPENED);
176 	
177 	this.animating = true;
178 	anim.animate();
179 	
180 	return true;
181 };

184 /**
185  * Hides an item
186  * @param {string} id the id of the item to hide
187  * @returns {boolean} true if close occurred, false otherwise
188  */
189 Exanpe.Accordion.prototype.close = function(id){
190 	if(this.animating){
191 		return false;
192 	}
193 	
194 	if(!this.isOpened(id)){
195 		return false;
196 	}
197 	
198 	if(!this.multiple && this.currentItem != id){
199 		return false;
200 	}
201 		
202 	var contentEl = this.getContentEl(id);
203 	
204 	var height = this.getContentElHeight(contentEl);
205 	
206 	var acc = this;
207 	var anim = new YAHOO.util.Anim(contentEl, Exanpe.Accordion.HIDE_ANIM_ATT, this.duration);
208 	anim.onComplete.subscribe(function(){
209 		YAHOO.util.Dom.removeClass(id, Exanpe.Accordion.CSS_OPENED);
210 		YAHOO.util.Dom.addClass(id, Exanpe.Accordion.CSS_CLOSED);
211 		YAHOO.util.Dom.setStyle(contentEl,"height",height+"px");
212 		acc.animating = false;
213 	});
214 
215 	this.animating = true;
216 	anim.animate();
217 	
218 	this.currentItem = null;
219 	
220 	return true;
221 };

302 /** 
303  * Constructor
304  * @class Represents an Hideable Panel. Can be accessed through JavaScript by its id.
305  * @param {String} id the id of the main accordion element
306  * @param {int} hideWidth the width of the hideable panel
307  * @param {float} duration the speed of hidding animation
308  */
309 Exanpe.HideablePanel = function(id, hideWidth, duration){
310 	/**
311 	 * id of the instance
312 	 */
313 	this.id = id;
314 	
315 	/**
316 	 * duration of the animation
317 	 */
318 	this.duration = duration;
319 	
320 	/**
321 	 * state of the panel
322 	 */
323 	this.hidden = false;
324 	
325 	var attributesHide = { 
326         width: { to: 0 }
327     }; 
328 	this.animHide = new YAHOO.util.Anim(this.id+'_hidepart', attributesHide, this.duration);
329 	this.animHide.onComplete.subscribe( function(){
330 		YAHOO.util.Dom.addClass(id, 'hidden');
331 	});
332 	
333 	var attributesShow = { 
334         width: { to: hideWidth  ,unit: 'px'}
335     }; 
336 	this.animShow = new YAHOO.util.Anim(this.id+'_hidepart', attributesShow, this.duration); 
337 	this.animShow.onComplete.subscribe( function(){
338 		YAHOO.util.Dom.removeClass(id, 'hidden');
339 	});
340 	
341 	var hideBar = this.getHideBarEl();
342 	
343 	YAHOO.util.Event.addListener(hideBar, "click", Exanpe.HideablePanel.prototype.reverse, null, this);
344     YAHOO.util.Event.addListener(hideBar, "mouseover", function(ev, id){YAHOO.util.Dom.addClass(hideBar, 'hidebarHover');}, this.id);
345     YAHOO.util.Event.addListener(hideBar, "mouseout", function(ev, id){YAHOO.util.Dom.removeClass(hideBar, 'hidebarHover');}, this.id);
346 };

370 		this.afterHide();
371 		return false;
372 	}	
373 };

479 /**
480  * Method triggered on hover start
481  * @param {Event} e the event of hovering
482  * @private
483  */
484 Exanpe.SecurePassword.prototype._hover = function(e){
485 
486 	var value = this.positionnings[YAHOO.util.Event.getTarget(e).id];
487 	
488 	YAHOO.util.Event.addListener(YAHOO.util.Event.getTarget(e).id, "mouseout", Exanpe.SecurePassword.prototype._cancelHover, this, true);
489 	if(this.timeoutHover){
490 		this._cancelHover();
491 	}
492 	
493 	var obj = this;
494 	
495 	this.timeoutHover = window.setTimeout(function(){
496 		obj._processRecording(value);
497 		obj = null;
498 		value = null;
499 	}, Exanpe.SecurePassword.HOVER_WAIT);	
500 	
501 };

508 Exanpe.SecurePassword.prototype._click = function(e){

514 /**
515  * Method actually performing the record of a value
516  * @param {char} value the character to record
517  * @private
518  */
519 Exanpe.SecurePassword.prototype._processRecording = function(value){
520 	var inputEl = this.getInputEl();
521 	var pwdEl = this.getHiddenEl();
522 	
523 	if(!inputEl.value){
524 		inputEl.value = value;
525 		pwdEl.value = inputEl.value;
526 		return;
527 	}
528 	
529 	if(inputEl.size && inputEl.value.length >= inputEl.size){
530 		return;
531 	}
532 	
533 	inputEl.value = inputEl.value + value;
534 	pwdEl.value = inputEl.value;
535 	pwdEl = null;
536 	inputEl = null;
537 };

602  * @param {String} zone the Tapestry zone element to update at the end of the ajax request

604 Exanpe.Slider = function(id, min, max, displayId, ajax, url, zone){
605 	/**

622 	 */
623 	this.displayId = displayId;
624 	
625 	/**
626 	 * Pixels number for 1 unit
627 	 */
628 	this.unit = Exanpe.Slider.LENGTH/(this.max-this.min);

630 	/**

633 	this.yui = null;

640 	/**

642 	 */
643 	this.url = url;
644 	
645 	/**
646 	 * Used to update a Tapestry zone element
647 	 */
648 	this.zone = zone;
649 	
650 	/**

652 	 */
653 	this.request = null;

713 /**
714  * Set the value of the slider.
715  * @param {int} value the value to set
716  * @param {Boolean} skipSlider whether the slider value update should me skipped. Set to true if the value initially comes from the YUI component.
717  */
718 Exanpe.Slider.prototype.setValue = function(value, skipSlider) {
719 	if(value < this.min || value > this.max){
720 		Exanpe.Log.error("Incorrect value set:"+value+". Min:"+this.min+"/Max:"+this.max);
721 		return ;
722 	}
723 	if(this._getDisplayEl()){
724 		this._getDisplayEl().innerHTML = ""+value;
725 	}
726 	if(skipSlider !== true){
727 		this.yui.setValue(this._valueToPixel(value), true, false, true);
728 	}
729 	this.value = value;
730 	this._getHiddenEl().value = value;
731 };

741 /**
742  * Convert a value to a yui pixel value.
743  * @param {int} value the value to set
744  * @return a value to the pixel unit of YUI
745  * @private
746  */
747 Exanpe.Slider.prototype._valueToPixel = function(value) {
748 	var zero = (-this.min)*this.unit;
749 	return zero + value*this.unit;

756  * @private

759 	var zero = (-this.min)*this.unit;
760 	return (value - zero)/this.unit;

815 	var request = YAHOO.util.Connect.asyncRequest(
